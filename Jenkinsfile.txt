pipeline {
	agent any
	stages {
		stage("Parallel Execution"){
			steps{
				parallel{
					a: {
					bat "mvn clean"
					},
					b: {
					bat "mvn test"
					},
					c: {
					bat "mvn package"
					},
				}
			}
		}

		stage("Consolidate Result"){
		steps{
			input("Do u want to Capture the Result?")
			junit'**/target/surefire-reports/TEST-*.xml'
			archive 'target/*.jar'
			}
		
		
		}
		
		
	}
}
