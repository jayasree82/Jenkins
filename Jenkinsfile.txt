pipeline {
	agent any
	stages {
		stage("Cleaning Stage"){
			steps{
				bat "mvn clean"
			}
		}
		stage("Testing Stage"){
			steps{
				bat "mvn test"
			}
		}
		stage("Packaging Stage"){
			steps{
				bat "mvn package"
			}
		
		}
		stage("Consolidate Result"){
		steps{
			input("Do u want to Capture the Result?")
			junit'**/target/surefire-reports/TEST-*.xml'
			archive 'target/*.jar'
			}
		
		
		}
		
		stage("Email Build Status"){
			steps{
mail body: '${env.JOB_NAME}-Build#${env.BUILD_NUMBER}-${currentBuild.currentResult}\n\nCheck Console output at ${env.BUILD_URL}to view the results', subject: '${env.JOB_NAME}-Build#${env.BUILD_NUMBER}-${currentBuild.currentResult}!!', to: 'er.jayasree@yahoo.com'  
		
	}
}
}
}
